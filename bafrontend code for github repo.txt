function fetchStudentData() {
    document.getElementById('studentCalendarContainer').innerHTML = "<p class='text-center p-4'>Opening Dairy...</p>";
    google.script.run.withSuccessHandler(tasks => {
        // Group tasks by date key
        studentTasksMap = {};
        let overdueCount = 0;
        const today = new Date().setHours(0,0,0,0);

        tasks.forEach(t => {
            const dateKey = new Date(t.scheduleDate).toISOString().split('T')[0];
            if (!studentTasksMap[dateKey]) studentTasksMap[dateKey] = [];
            studentTasksMap[dateKey].push(t);

            if (new Date(t.scheduleDate).getTime() < today && t.isDone !== "Yes") {
                overdueCount++;
            }
        });

        // Update Overdue Notification
        const badge = document.getElementById('studentOverdueCount');
        if (overdueCount > 0) {
            badge.innerText = overdueCount + " OVERDUE";
            badge.classList.remove('hidden');
        }

        renderStudentCalendar();
    }).getStudentTasks(currentUser.id);
}

function renderStudentCalendar() {
    const now = new Date();
    const year = now.getFullYear();
    const month = now.getMonth();
    const firstDay = new Date(year, month, 1).getDay();
    const daysInMonth = new Date(year, month + 1, 0).getDate();
    const monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
    
    let html = `<div class="text-center font-bold mb-4 text-orange-400">${monthNames[month]} ${year}</div>`;
    html += `<div class="calendar-grid">`;
    ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"].forEach(d => html += `<div class="calendar-header">${d}</div>`);
    
    // Fill empty start slots
    for (let i = 0; i < firstDay; i++) html += `<div class="calendar-day opacity-20"></div>`;

    // Fill days
    for (let day = 1; day <= daysInMonth; day++) {
        const dateKey = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
        const dayTasks = studentTasksMap[dateKey] || [];
        const allDone = dayTasks.length > 0 && dayTasks.every(t => t.isDone === "Yes");
        const hasPending = dayTasks.some(t => t.isDone !== "Yes");
        
        let dayClass = "";
        if (hasPending) dayClass = "border-l-4 border-orange-500";
        if (dayTasks.length > 0 && allDone) dayClass = "border-l-4 border-green-500";

        html += `
            <div class="calendar-day ${dayClass}" onclick="showStudentDayTasks('${dateKey}')">
                <span class="day-number ${hasPending ? 'text-orange-400 font-bold' : ''}">${day}</span>
                <div class="space-y-1">
                    ${dayTasks.map(t => `
                        <div style="font-size: 0.6rem;" class="truncate ${t.isDone === 'Yes' ? 'text-green-400 line-through opacity-50' : 'text-white'}">
                            • ${t.topic}
                        </div>
                    `).join('')}
                </div>
            </div>`;
    }
    
    document.getElementById('studentCalendarContainer').innerHTML = html + `</div>`;
}

function showStudentDayTasks(dateKey) {
    const tasks = studentTasksMap[dateKey];
    const container = document.getElementById('studentDayDetail');
    const list = document.getElementById('studentTaskList');
    
    if (!tasks || tasks.length === 0) {
        container.classList.add('hidden');
        return;
    }

    document.getElementById('studentSelectedDate').innerText = new Date(dateKey).toLocaleDateString(undefined, {dateStyle:'full'});
    container.classList.remove('hidden');
    
    let html = "";
    tasks.forEach(t => {
        const isDone = t.isDone === "Yes";
        html += `
            <div class="flex items-center mb-4 p-2 bg-white bg-opacity-5 rounded">
                <input type="checkbox" ${isDone ? "checked" : ""} 
                    onchange="updateTaskStatus('${t.id}', this.checked, '${dateKey}')" 
                    class="h-6 w-6 mr-4 accent-orange-500 cursor-pointer">
                <span class="text-lg ${isDone ? 'task-done' : ''}">${t.topic}</span>
            </div>`;
    });
    list.innerHTML = html;
    container.scrollIntoView({behavior: 'smooth'});
}

function updateTaskStatus(taskId, isChecked, dateKey) {
    // 1. Update Backend
    google.script.run.toggleTask(taskId, isChecked);
    
    // 2. Update Local Map
    if (studentTasksMap[dateKey]) {
        const task = studentTasksMap[dateKey].find(t => t.id === taskId);
        if (task) task.isDone = isChecked ? "Yes" : "No";
    }
    
    // 3. Refresh Visuals
    renderStudentCalendar();
}

        function renderCalendar() {
            const now = new Date();
            const year = now.getFullYear();
            const month = now.getMonth();
            const firstDay = new Date(year, month, 1).getDay();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            const monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
            
            let html = `<div class="text-center font-bold mb-2">${monthNames[month]} ${year}</div>`;
            html += `<div class="calendar-grid">`;
            ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"].forEach(d => html += `<div class="calendar-header">${d}</div>`);
            
            for (let i = 0; i < firstDay; i++) html += `<div class="calendar-day opacity-20"></div>`;

            for (let day = 1; day <= daysInMonth; day++) {
                const dateKey = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
                const tasks = globalCalendarData[dateKey] || [];
                const doneCount = tasks.filter(t => t.isDone).length;
                
                html += `<div class="calendar-day" onclick="showDayDetail('${dateKey}')">
                    <span class="day-number">${day}</span>
                    <div class="flex flex-wrap">${tasks.slice(0, 5).map(t => `<span class="dot-indicator ${t.isDone ? 'bg-green-500' : 'bg-orange-500'}"></span>`).join('')}</div>
                    ${tasks.length > 0 ? `<div style="font-size:0.65rem" class="mt-1">${doneCount}/${tasks.length} Done</div>` : ''}
                </div>`;
            }
            document.getElementById('calendarContainer').innerHTML = html + `</div>`;
        }

        function showDayDetail(dateKey) {
            const tasks = globalCalendarData[dateKey];
            if (!tasks) return;
            const container = document.getElementById('dayDetailContainer');
            document.getElementById('selectedDateTitle').innerText = "Revisions: " + dateKey;
            container.classList.remove('hidden');
            
            let h = `<table class="w-full text-xs text-left"><thead><tr class="opacity-70 border-b"><th>Student</th><th>Topic</th><th>Status</th></tr></thead><tbody>`;
            tasks.forEach(t => {
                h += `<tr class="border-b border-gray-700"><td class="py-2">${t.studentName}</td><td>${t.topic}</td><td class="${t.isDone ? 'text-green-400' : 'text-orange-400'}">${t.isDone ? 'Done' : 'Pending'}</td></tr>`;
            });
            document.getElementById('dayDetailList').innerHTML = h + `</tbody></table>`;
            container.scrollIntoView({behavior: 'smooth'});
        }

        function editUser(u) {
            const name = prompt("Name", u.name || "");

            const loginId = prompt("Login ID", u.loginid || "");
            const password = prompt("Password", u.password || "");
            const role = prompt("Role (student/teacher/admin)", u.role || "student");

            if(role === 'student') { 
                c = prompt("Class (1:Class 10 ... 13:Jr Kg)", u.classindex || "1"); 
                b = prompt("Board (1:Guj, 2:CBSE, 3:ICSE)", u.board || "1");

            google.script.run.withSuccessHandler(() => showAdminSection('users')).adminSaveUser({ id: u.id, loginId, password, name, role, classIndex: c, board: b, isLocked: lock });
